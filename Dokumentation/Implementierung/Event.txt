-- Table: event

-- DROP TABLE event;

CREATE TABLE event
(
  eventid bigint NOT NULL,
  creatorid bigint,
  eventuserid bigint,
  date date,
  eventname text,
  occasion text,
  location text,
  gps character varying[],
  description text,
  numbermaleconfirmed bigint,
  numberfemaleconfirmed bigint,
  CONSTRAINT eventid PRIMARY KEY (eventid )
)
WITH (
  OIDS=FALSE
);
ALTER TABLE event
  OWNER TO postgres;
