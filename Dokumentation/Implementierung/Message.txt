CREATE TABLE message
(
  messageid bigint NOT NULL,
  eventid bigint,
  senderid bigint,
  recipientid bigint,
  textmessage text,
  date date,
  CONSTRAINT messageid PRIMARY KEY (messageid )
)
WITH (
  OIDS=FALSE
);
